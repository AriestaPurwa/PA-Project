        @include('projects.partials.category-node',

@push('scripts')
<script>
document.addEventListener('click', function(e) {

    const title = e.target.closest('.category-title');
    if (!title) return;

    const nested = title.parentElement.querySelector('.nested');
    const arrow = title.querySelector('.arrow');

    if (!nested) return;

    nested.classList.toggle('active');

    arrow.textContent =
        nested.classList.contains('active')
        ? '▼'
        : '▶';
});
console.log("RBS SCRIPT LOADED");
</script>
@endpush
